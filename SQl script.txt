DROP TABLE IF EXISTS Weapon CASCADE;
DROP TABLE IF EXISTS Vehicule CASCADE;
DROP TABLE IF EXISTS Location CASCADE;
DROP TABLE IF EXISTS Description CASCADE;
DROP TABLE IF EXISTS Accident CASCADE;

CREATE TABLE IF NOT EXISTS Weapon(
	WeaponId UUID primary key,
	Name varchar(50),
	Builder varchar(20),
	Description varchar(2000)
);

CREATE TABLE IF NOT EXISTS Vehicule(
	VehiculeId UUID primary key,
	Type varchar(50),
	Builder varchar(20),
	Name varchar(50),
	Description varchar(3000)
);

CREATE TABLE IF NOT EXISTS Location(
	LocationId UUID primary key,
	Country varchar(50),
	PositionLost varchar(100),
	XCoordonate float,
	YCoordonate float
);


CREATE TABLE IF NOT EXISTS Accident (
    AccidentId UUID PRIMARY KEY,
    LocationId UUID,
    VehiculeId UUID,
    WeaponId UUID,
	seal varchar(10),
	isBrokenArrow BOOLEAN,
	isCloseCall BOOLEAN,
    DisasterDate DATE,
    ShortDescription VARCHAR(2000),
    BubbleDescription VARCHAR(500),
	FullDescription VARCHAR(2000),
    CONSTRAINT fk_country FOREIGN KEY (LocationId) REFERENCES LOCATION(LocationId),
    CONSTRAINT fk_vehicule FOREIGN KEY (VehiculeId) REFERENCES VEHICULE(VehiculeId),
    CONSTRAINT fk_weapon FOREIGN KEY (WeaponId) REFERENCES WEAPON(WeaponId)
);
